211 class Timezone:
212 """
213 Timezone information returned by PytzCache.__getitem__
214 Adapts datetime.tzinfo object to DateTime._timezone interface
215 """
216 def __init__(self, tzinfo):
217 self.tzinfo = tzinfo
218  
219 def info(self, t=None):
220 if t is None:
221 dt = datetime.utcnow().replace(tzinfo=pytz.utc)
222 else:
223 # can't use utcfromtimestamp past 2038
224 dt = EPOCH + timedelta(0, t)
225  
226 # need to normalize tzinfo for the datetime to deal with
227 # daylight savings time.
228 normalized_dt = self.tzinfo.normalize(dt.astimezone(self.tzinfo))
229 normalized_tzinfo = normalized_dt.tzinfo
230  
231 offset = normalized_tzinfo.utcoffset(normalized_dt)
232 secs = offset.days * 24 * 60 * 60 + offset.seconds
233 dst = normalized_tzinfo.dst(normalized_dt)
234 if dst == timedelta(0):
235 is_dst = 0
236 else:
237 is_dst = 1
238 return secs, is_dst, normalized_tzinfo.tzname(normalized_dt)
