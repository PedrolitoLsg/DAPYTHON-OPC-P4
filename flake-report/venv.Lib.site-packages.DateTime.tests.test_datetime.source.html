70 def _compare(self, dt1, dt2):
71 '''Compares the internal representation of dt1 with
72 the representation in dt2. Allows sub-millisecond variations.
73 Primarily for testing.'''
74 self.assertEqual(round(dt1._t, 3), round(dt2._t, 3))
75 self.assertEqual(round(dt1._d, 9), round(dt2._d, 9))
76 self.assertEqual(round(dt1.time, 9), round(dt2.time, 9))
77 self.assertEqual(dt1.millis(), dt2.millis())
78 self.assertEqual(dt1._micros, dt2._micros)
