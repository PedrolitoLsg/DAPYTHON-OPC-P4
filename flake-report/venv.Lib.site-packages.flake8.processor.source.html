188 def build_logical_line_tokens(self): # type: () -> _Logical
189 """Build the mapping, comments, and logical line lists."""
190 logical = []
191 comments = []
192 mapping = [] # type: _LogicalMapping
193 length = 0
194 previous_row = previous_column = None
195 for token_type, text, start, end, line in self.tokens:
196 if token_type in SKIP_TOKENS:
197 continue
198 if not mapping:
199 mapping = [(0, start)]
200 if token_type == tokenize.COMMENT:
201 comments.append(text)
202 continue
203 if token_type == tokenize.STRING:
204 text = mutate_string(text)
205 if previous_row:
206 (start_row, start_column) = start
207 if previous_row != start_row:
208 row_index = previous_row - 1
209 column_index = previous_column - 1
210 previous_text = self.lines[row_index][column_index]
211 if previous_text == "," or (
212 previous_text not in "{[(" and text not in "}])"
213 ):
214 text = " " + text
215 elif previous_column != start_column:
216 text = line[previous_column:start_column] + text
217 logical.append(text)
218 length += len(text)
219 mapping.append((length, end))
220 (previous_row, previous_column) = end
221 return comments, logical, mapping

461 def expand_indent(line): # type: (str) -> int
462 r"""Return the amount of indentation.
463  
464 Tabs are expanded to the next multiple of 8.
465  
466 >>> expand_indent(' ')
467 4
468 >>> expand_indent('\t')
469 8
470 >>> expand_indent(' \t')
471 8
472 >>> expand_indent(' \t')
473 16
474 """
475 if "\t" not in line:
476 return len(line) - len(line.lstrip())
477 result = 0
478 for char in line:
479 if char == "\t":
480 result = result // 8 * 8 + 8
481 elif char == " ":
482 result += 1
483 else:
484 break
485 return result

491 def mutate_string(text): # type: (str) -> str
492 """Replace contents with 'xxx' to prevent syntax matching.
493  
494 >>> mutate_string('"abc"')
495 '"xxx"'
496 >>> mutate_string("'''abc'''")
497 "'''xxx'''"
498 >>> mutate_string("r'abc'")
499 "r'xxx'"
500 """
501 # NOTE(sigmavirus24): If there are string modifiers (e.g., b, u, r)
502 # use the last "character" to determine if we're using single or double
503 # quotes and then find the first instance of it
504 start = text.index(text[-1]) + 1
505 end = len(text) - 1
506 # Check for triple-quoted strings
507 if text[-3:] in ('"""', "'''"):
508 start += 2
509 end -= 2
510 return text[:start] + "x" * (end - start) + text[end:]
