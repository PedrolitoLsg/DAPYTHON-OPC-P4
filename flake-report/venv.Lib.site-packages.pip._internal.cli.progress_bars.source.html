52 class InterruptibleMixin:
53 """
54 Helper to ensure that self.finish() gets called on keyboard interrupt.
55  
56 This allows downloads to be interrupted without leaving temporary state
57 (like hidden cursors) behind.
58  
59 This class is similar to the progress library's existing SigIntMixin
60 helper, but as of version 1.2, that helper has the following problems:
61  
62 1. It calls sys.exit().
63 2. It discards the existing SIGINT handler completely.
64 3. It leaves its own handler in place even after an uninterrupted finish,
65 which will have unexpected delayed effects if the user triggers an
66 unrelated keyboard interrupt some time after a progress-displaying
67 download has already completed, for example.
68 """
69  
70 def __init__(self, *args, **kwargs):
71 # type: (List[Any], Dict[Any, Any]) -> None
72 """
73 Save the original SIGINT handler for later.
74 """
75 # https://github.com/python/mypy/issues/5887
76 super().__init__(*args, **kwargs) # type: ignore
77  
78 self.original_handler = signal(SIGINT, self.handle_sigint)
79  
80 # If signal() returns None, the previous handler was not installed from
81 # Python, and we cannot restore it. This probably should not happen,
82 # but if it does, we must restore something sensible instead, at least.
83 # The least bad option should be Python's default SIGINT handler, which
84 # just raises KeyboardInterrupt.
85 if self.original_handler is None:
86 self.original_handler = default_int_handler

88 def finish(self):
89 # type: () -> None
90 """
91 Restore the original SIGINT handler after finishing.
92  
93 This should happen regardless of whether the progress display finishes
94 normally, or gets interrupted.
95 """

100 """

102  
103 This handler should only be in place while the progress display is
104 active.
105 """
106 self.finish()
