218 def get_installation_order(self, req_set):
219 # type: (RequirementSet) -> List[InstallRequirement]
220 """Get order for installation of requirements in RequirementSet.
221  
222 The returned list contains a requirement before another that depends on

224 get installed one-by-one.
225  
226 The current implementation creates a topological ordering of the

247 def get_topological_weights(graph, expected_node_count):
248 # type: (DirectedGraph[Optional[str]], int) -> Dict[Optional[str], int]
249 """Assign weights to each node based on how "deep" they are.
250  
251 This implementation may change at any point in the future without prior
252 notice.
253  
254 We take the length for the longest path to any node from root, ignoring any
255 paths that contain a single node twice (i.e. cycles). This is done through
256 a depth-first search through the graph, while keeping track of the path to
257 the node.
258  
259 Cycles in the graph result would result in node being revisited while also
260 being it's own path. In this case, take no action. This helps ensure we
261 don't get stuck in a cycle.
262  
263 When assigning weight, the longer path (i.e. larger length) is preferred.
264 """
265 path = set() # type: Set[Optional[str]]
266 weights = {} # type: Dict[Optional[str], int]
267  
268 def visit(node):
269 # type: (Optional[str]) -> None
270 if node in path:
271 # We hit a cycle, so we'll break it here.
272 return
273  
274 # Time to visit the children!
275 path.add(node)
276 for child in graph.iter_children(node):
277 visit(child)
278 path.remove(node)
279  
280 last_known_parent_count = weights.get(node, 0)
281 weights[node] = max(last_known_parent_count, len(path))
282  
283 # `None` is guaranteed to be the root node by resolvelib.
284 visit(None)
285  
286 # Sanity checks
287 assert weights[None] == 0
288 assert len(weights) == expected_node_count
289  
290 return weights
