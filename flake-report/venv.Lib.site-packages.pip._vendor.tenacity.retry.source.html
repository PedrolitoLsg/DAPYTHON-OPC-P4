175 class retry_any(retry_base):
176 """Retries if any of the retries condition is valid."""
177  
178 def __init__(self, *retries):
179 self.retries = retries
180  
181 def __call__(self, retry_state):
182 return any(r(retry_state) for r in self.retries)
