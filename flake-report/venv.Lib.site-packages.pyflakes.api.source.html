20 def check(codeString, filename, reporter=None):
21 """
22 Check the Python source given by C{codeString} for flakes.
23  
24 @param codeString: The Python source to check.
25 @type codeString: C{str}
26  
27 @param filename: The name of the file the source came from, used to report
28 errors.
29 @type filename: C{str}
30  
31 @param reporter: A L{Reporter} instance, where errors and warnings will be
32 reported.
33  
34 @return: The number of warnings emitted.
35 @rtype: C{int}
36 """
37 if reporter is None:
38 reporter = modReporter._makeDefaultReporter()
39 # First, compile into an AST and handle syntax errors.
40 try:
41 tree = ast.parse(codeString, filename=filename)
42 except SyntaxError:
43 value = sys.exc_info()[1]
44 msg = value.args[0]
45  
46 (lineno, offset, text) = value.lineno, value.offset, value.text
47  
48 if checker.PYPY:
49 if text is None:
50 lines = codeString.splitlines()
51 if len(lines) >= lineno:
52 text = lines[lineno - 1]
53 if sys.version_info >= (3, ) and isinstance(text, bytes):
54 try:
55 text = text.decode('ascii')
56 except UnicodeDecodeError:
57 text = None
58 offset -= 1
59  
60 # If there's an encoding problem with the file, the text is None.
61 if text is None:
62 # Avoid using msg, since for the only known case, it contains a
63 # bogus message that claims the encoding the file declared was
64 # unknown.
65 reporter.unexpectedError(filename, 'problem decoding source')
66 else:
67 reporter.syntaxError(filename, msg, lineno, offset, text)
68 return 1
69 except Exception:
70 reporter.unexpectedError(filename, 'problem decoding source')
71 return 1

73 file_tokens = checker.make_tokens(codeString)
74 w = checker.Checker(tree, file_tokens=file_tokens, filename=filename)
75 w.messages.sort(key=lambda m: m.lineno)
76 for warning in w.messages:
77 reporter.flake(warning)
78 return len(w.messages)

102 def isPythonFile(filename):
103 """Return True if filename points to a Python file."""
104 if filename.endswith('.py'):
105 return True
106  
107 # Avoid obvious Emacs backup files
108 if filename.endswith("~"):
109 return False
110  
111 max_bytes = 128
112  
113 try:
114 with open(filename, 'rb') as f:
115 text = f.read(max_bytes)

117 return False
118 except IOError:
119 return False

124 def iterSourceCode(paths):
125 """
126 Iterate over all Python source files in C{paths}.
127  
128 @param paths: A list of paths. Directories will be recursed into and
129 any .py files found will be yielded. Any non-directories will be
130 yielded as-is.
131 """
132 for path in paths:
133 if os.path.isdir(path):
134 for dirpath, dirnames, filenames in os.walk(path):
135 for filename in filenames:
136 full_path = os.path.join(dirpath, filename)
137 if isPythonFile(full_path):
138 yield full_path
139 else:
140 yield path

143 def checkRecursive(paths, reporter):
144 """
145 Recursively check all source files in C{paths}.
146  
147 @param paths: A list of paths to Python source files and directories
148 containing Python source files.
149 @param reporter: A L{Reporter} where all of the warnings and errors
150 will be reported to.
151 @return: The number of warnings found.
152 """
153 warnings = 0
154 for sourcePath in iterSourceCode(paths):
155 warnings += checkPath(sourcePath, reporter)
156 return warnings

159 def _exitOnSignal(sigName, message):
160 """Handles a signal with sys.exit.
161  
162 Some of these signals (SIGPIPE, for example) don't exist or are invalid on
163 Windows. So, ignore errors that might arise.
164 """
165 import signal
166  
167 try:
168 sigNumber = getattr(signal, sigName)
169 except AttributeError:
170 # the signal constants defined in the signal module are defined by
171 # whether the C library supports them or not. So, SIGPIPE might not
172 # even be defined.
173 return

176 sys.exit(message)
177  
178 try:

180 except ValueError:
