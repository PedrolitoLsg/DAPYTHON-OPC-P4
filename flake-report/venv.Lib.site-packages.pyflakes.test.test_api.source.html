63 class SysStreamCapturing(object):
64  
65 """
66 Context manager capturing sys.stdin, sys.stdout and sys.stderr.
67  
68 The file handles are replaced with a StringIO object.
69 On environments that support it, the StringIO object uses newlines
70 set to os.linesep. Otherwise newlines are converted from \\n to
71 os.linesep during __exit__.
72 """
73  
74 def _create_StringIO(self, buffer=None):
75 # Python 3 has a newline argument
76 try:
77 return StringIO(buffer, newline=os.linesep)
78 except TypeError:
79 self._newline = True
80 # Python 2 creates an input only stream when buffer is not None
81 if buffer is None:
82 return StringIO()
83 else:
84 return StringIO(buffer)
85  
86 def __init__(self, stdin):
87 self._newline = False
88 self._stdin = self._create_StringIO(stdin or '')
89  
90 def __enter__(self):
91 self._orig_stdin = sys.stdin
92 self._orig_stdout = sys.stdout
93 self._orig_stderr = sys.stderr
94  
95 sys.stdin = self._stdin
96 sys.stdout = self._stdout_stringio = self._create_StringIO()
97 sys.stderr = self._stderr_stringio = self._create_StringIO()
98  
99 return self
100  
101 def __exit__(self, *args):
102 self.output = self._stdout_stringio.getvalue()
103 self.error = self._stderr_stringio.getvalue()
104  
105 if self._newline and os.linesep != '\n':
106 self.output = self.output.replace('\n', os.linesep)
107 self.error = self.error.replace('\n', os.linesep)
108  
109 sys.stdin = self._orig_stdin
110 sys.stdout = self._orig_stdout
111 sys.stderr = self._orig_stderr
