26 def doctestify(self, input):
27 lines = []
28 for line in textwrap.dedent(input).splitlines():
29 if line.strip() == '':
30 pass
31 elif (line.startswith(' ') or
32 line.startswith('except:') or
33 line.startswith('except ') or
34 line.startswith('finally:') or
35 line.startswith('else:') or
36 line.startswith('elif ') or
37 (lines and lines[-1].startswith(('>>> @', '... @')))):
38 line = "... %s" % line
39 else:
40 line = ">>> %s" % line
41 lines.append(line)
42 doctestificator = textwrap.dedent('''\
43 def doctest_something():
44 """
45 %s
46 """
47 ''')
48 return doctestificator % "\n ".join(lines)
