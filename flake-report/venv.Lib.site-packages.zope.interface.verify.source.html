182 def _pypy2_false_positive(msg, candidate, vtype):
183 # On PyPy2, builtin methods and functions like

185 # (those with no default, something you can't express in Python 2
186 # syntax; CPython uses special internal APIs to implement these methods)
187 # return false failures because PyPy2 doesn't expose any way
188 # to detect this pseudo-optional status. PyPy3 doesn't have this problem
189 # because of __defaults_count__, and CPython never gets here because it
190 # returns true for ``ismethoddescriptor`` or ``isbuiltin``.
191 #
192 # We can't catch all such cases, but we can handle the common ones.
193 #

195 return False
